void LCD_fillSemiCircle(int x0, int y0, int r, UINT8 cornername,int delta)
{
	int f     = 1 - r;
	int ddF_x = 1;
	int ddF_y = -2 * r;
	int x     = 0;
	int y     = r;

	while (x<y) 
	{
    	if (f >= 0) 
		{
      		y--;
      		ddF_y += 2;
      		f += ddF_y;
    	}
    	x++;
    	ddF_x += 2;
    	f += ddF_x;

    	if (cornername & 0x1) 
		{
      		LCD_fill(x0+x, y0-y,x0+x, y0+y+1+delta);
      		LCD_fill(x0+y, y0-x,x0+y, y0+x+1+delta);
    	}
    	if (cornername & 0x2) 
		{
      		LCD_fill(x0-x, y0-y, x0-x, y0+y+1+delta);
      		LCD_fill(x0-y, y0-x, x0-y, y0+x+1+delta);
    	}
  	}
}

/******************************************************************************
      ÂáΩÊï∞ËØ¥ÊòéÔºöÁîªÁ∫ø

void LCD_drawLine(UINT16 x0,UINT16 y0,UINT16 x1,UINT16 y1)
{
	UINT16 t; 
	int xErr=0,yErr=0,dx,dy,dist;
	int incX,incY,x,y;
	dx=x1-x0; //ËÆ°ÁÆóÂùêÊ†áÂ¢ûÈáè 
	dy=y1-y0;
	x=x0;//ÁîªÁ∫øËµ∑ÁÇπÂùêÊ†á
	y=y0;
	if(dx>0)incX=1; //ËÆæÁΩÆÂçïÊ≠•ÊñπÂêë 
	else if (dx==0)incX=0;//ÂûÇÁõ¥Á∫ø 
	else {incX=-1;dx=-dx;}
	if(dy>0)incY=1;
	else if (dy==0)incY=0;//Ê∞¥Âπ≥Á∫ø 
	else {incY=-1;dy=-dy;}
	if(dx>dy)dist=dx; //ÈÄâÂèñÂü∫Êú¨Â¢ûÈáèÂùêÊ†áËΩ¥ 
	else dist=dy;
	for(t=0;t<dist+1;t++)
	{
		LCD_drawPixel(x,y);//ÁîªÁÇπ
		xErr+=dx;
		yErr+=dy;
		if(xErr>dist)
		{
			xErr-=dist;
			x+=incX;
		}
		if(yErr>dist)
		{
			yErr-=dist;
			y+=incY;
		}
	}
}
