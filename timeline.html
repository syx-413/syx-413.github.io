时钟时序线

Posted on Sat, Feb 17, 2024 单片机

缘由

突发有一天想要一个高质量的时序图来表达自己的想法，就找到了一个开源的方案.

WAVEDROM

效果很好，语法简单，易于上手

信号的不同种类

可以镶嵌函数代替

(function (bits, ticks) {
  var i, t, gray, state, data = [], arr = [];
  for (i = 0; i < bits; i++) {
    arr.push({name: i + '', wave: ''});
    state = 1;
    for (t = 0; t < ticks; t++) {
      data.push(t + '');
      gray = (((t >> 1) ^ t) >> i) & 1;
      arr[i].wave += (gray === state) ? '.' : gray + '';
      state = gray;
    }
  }
  arr.unshift('gray');
  return {signal: [
    {name: 'bin', wave: '='.repeat(ticks), data: data}, arr
  ]};
})(5, 16)
